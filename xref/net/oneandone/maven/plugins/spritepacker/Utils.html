21      /**
22       * Check if any input is newer than any output. This also takes output files into account that do not yet exist,
23       * as they have a lastModified time of 0L and are thus automatically "older" than any input files.
24       *
25       * @param inputs    the list of input files, must not be null
26       * @param outputs   the list of output files, must not be null
27       * @return          whether any input file was newer than any output file
28       */
29      public static boolean shouldWriteOutput(List<Path> inputs, List<Path> outputs) throws IOException {
30          Objects.requireNonNull(inputs);
31          Objects.requireNonNull(outputs);
32  
33          boolean hasNoInput = true;
34          long newestInput = 0L;
35          for (Path input : inputs) {
36              if (input != null && Files.exists(input)) {
37                  newestInput = Math.max(Files.getLastModifiedTime(input).toMillis(), newestInput);
38                  hasNoInput = false;
39              }
40          }
41  
42          if (hasNoInput) {
43              return false;
44          }
45  
46          for (Path output : outputs) {
47              if (output != null && (Files.notExists(output) || Files.getLastModifiedTime(output).toMillis() <= newestInput)) {
48                  return true;
49              }
50          }
51  
52          return false;
53      }
54  }
