8   import java.awt.Point;
9   import java.io.File;
10  
11  /**
12   * Converts ImagePacking to a LESS file, with the result that each icon's properties are available
13   * via mixins. the ".create" mixin returns all properties, the ".pos" mixin returns only the position
14   * and the ".size" mixin returns only the dimensions of the icon.
15   *
16   * @author mklein
17   */
18  public class LessPackingConverter extends AbstractTextConverter {
19      private final String lessNamespace;
20  
21      /**
22       * Create a LESS converter with output file less and namespace lessNamespace.
23       *
24       * @param less          the output LESS file to write to
25       * @param lessNamespace the LESS namespace under which the mixins should be added
26       */
27      public LessPackingConverter(File less, String lessNamespace) {
28          super(less, "LESS");
29          this.lessNamespace = lessNamespace;
30      }
31  
32      /**
33       * Create output LESS string based on an ImagePacking.
34       *
35       * @param imagePacking  the ImagePacking to convert
36       * @param log           the log object to use
37       * @return              String containing the LESS file contents
38       * @throws MojoExecutionException
39       */
40      @Override
41      protected String createOutput(ImagePacking imagePacking, Log log) throws MojoExecutionException {
42          StringBuilder sb = new StringBuilder("/* this file is generated by the sprite packer. don't make any changes in here! */\n")
43                                  .append("/* icons can be referenced with the syntax \"#")
44                                  .append(lessNamespace).append(" > .create(icon-name);\" */\n")
45                                  .append("#").append(lessNamespace).append("{\n")
46                                  .append(".create(@name){.pos(@name);.size(@name);}\n");
47          for (NamedImage image : imagePacking.getImages()) {
48              String name = image.getName();
49              Point position = imagePacking.getPosition(image);
50              String x = intToPixel(-position.x);
51              String y = intToPixel(-position.y);
52              String width = intToPixel(image.getWidth());
53              String height = intToPixel(image.getHeight());
54              sb.append("    .pos(").append(name).append("){background-position:").append(x).append(" ").append(y).append(";}\n")
55                .append("    .size(").append(name).append("){width:").append(width).append(";height:").append(height).append(";}\n");
56          }
57          sb.append("}");
58          return sb.toString();
59      }
60  }
