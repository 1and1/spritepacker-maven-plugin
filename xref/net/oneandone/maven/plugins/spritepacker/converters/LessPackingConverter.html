9   import java.awt.Point;
10  import java.nio.file.Path;
11  import java.util.List;

13  /**
14   * Converts ImagePacking to a Less file, with the result that each icon's properties are available
15   * via mixins. the ".create" mixin returns all properties, the ".pos" mixin returns only the position
16   * and the ".size" mixin returns only the dimensions of the icon.
17   *
18   * @author mklein
19   */
20  public class LessPackingConverter extends AbstractTextConverter {
21      final String lessNamespace;
22  
23      /**
24       * Create a Less converter with output file less and namespace lessNamespace.
25       * @param less          the output Less file to write to
26       * @param lessNamespace the Less namespace under which the mixins should be added
27       */
28      public LessPackingConverter(Path less, String lessNamespace) {
29          super(less, "Less");
30          this.lessNamespace = fixFirstChar(sanitize(lessNamespace));
31      }
32  
33      /**
34       * Create output Less string based on an ImagePacking.
35       *
36       * @param imageList     the list of images - must not be null
37       * @param imagePacking  the ImagePacking to convert - must not be null
38       * @param log           the log object to use
39       * @return              String containing the Less file contents
40       * @throws MojoExecutionException
41       */
42      @Override
43      protected String createOutput(List<NamedImage> imageList, ImagePacking imagePacking, Log log) throws MojoExecutionException {
44          StringBuilder sb = new StringBuilder("/* this file is generated by the sprite packer. don't make any changes in here! */\n");
45          String tab = "";
46          if (StringUtils.isNotEmpty(lessNamespace)) {
47              tab = "    ";
48              sb.append("/* icons can be referenced with the syntax \"#").append(lessNamespace).append(" > .create(icon-name);\" */\n")
49                .append("#").append(lessNamespace).append("{\n");
50          }
51          sb.append(tab).append(".create(@name){.pos(@name);.size(@name);}\n");
52          for (NamedImage image : imageList) {
53              String name = sanitize(image.getName());
54              Point position = imagePacking.getPosition(image);
55              String x = intToPixel(-position.x);
56              String y = intToPixel(-position.y);
57              String width = intToPixel(image.getWidth());
58              String height = intToPixel(image.getHeight());
59              sb.append(tab).append(".pos(").append(name).append("){background-position:").append(x).append(" ").append(y).append(";}\n")
60                .append(tab).append(".size(").append(name).append("){width:").append(width).append(";height:").append(height).append(";}\n");
61          }
62  
63          if (StringUtils.isNotEmpty(lessNamespace)) {
64              sb.append("}");
65          }
66          return sb.toString();
67      }
68  }
