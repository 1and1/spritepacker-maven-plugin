8   import java.io.IOException;
9   import java.nio.charset.StandardCharsets;
10  import java.nio.file.Files;
11  import java.nio.file.Path;
12  import java.util.List;
13  import java.util.regex.Pattern;
14  
15  /**
16   * An abstract PackingConverter that saves the output of the subclass's implemented createOutput method to a text file.
17   *
18   * @author mklein
19   */
20  public abstract class AbstractTextConverter implements PackingConverter {
21      public static final Pattern CHARS_NOT_ALLOWED_IN_IDENTIFIERS = Pattern.compile("[^\\p{L}\\p{N}_-]");
22      private final Path file;
23      private final String type;
24  
25      /**
26       * Create an AbstractTextConverter that saves to the specified file, and specify the type of output file for logging purposes.
27       * @param file  the output file to write to
28       * @param type  the type of file to convert to, for logging purposes
29       */
30      protected AbstractTextConverter(Path file, String type) {
31          this.file = file;
32          this.type = type;
33      }
34  
35      /**
36       * Convert the specified ImagePacking to a text file.
37       *
38       * @param imageList     the list of images
39       * @param imagePacking  the ImagePacking to convert
40       * @param log           the log object to use
41       * @throws MojoExecutionException
42       */
43      @Override
44      public void convert(List<NamedImage> imageList, ImagePacking imagePacking, Log log) throws MojoExecutionException {
45          if (file == null) {
46              log.info("No " + type + " output file specified.");
47              return;
48          }
49  
50          log.info("Generating " + type + " output...");
51  
52          String output = createOutput(imageList, imagePacking, log);
53  
54          try {
55              log.info("Saving " + type + " to file " + file.toAbsolutePath());
56              Files.write(file, output.getBytes(StandardCharsets.UTF_8));
57          } catch (IOException e) {
58              throw new MojoExecutionException("Couldn't write to file " + file.toAbsolutePath(), e);
59          }
60      }
61  
62      /**
63       * Create output text as String based on an ImagePacking.
64       *
65       * @param imageList     the list of images
66       * @param imagePacking  the ImagePacking to convert
67       * @param log           the log object to use
68       * @return              String containing the text file contents
69       * @throws MojoExecutionException
70       */
71      protected abstract String createOutput(List<NamedImage> imageList, ImagePacking imagePacking, Log log) throws MojoExecutionException;
72  
73      /**
74       * Convert an integer into a String pixel value, leaving off the "px" for values of 0.
75       *
76       * @param i     the int value to convert
77       * @return      the pixel value
78       */
79      protected static String intToPixel(int i) {
80          return i == 0 ? "0" : i + "px";
81      }
82  
83      /**
84       * Sanitize name for use in CSS or Less by removing all characters that are not letters, numbers
85       * hyphens or underscores.
86       *
87       * @param name  the name to be sanitized
88       * @return      sanitized name without special characters
89       */
90      protected static String sanitize(String name) {
91          return (name == null) ? null : CHARS_NOT_ALLOWED_IN_IDENTIFIERS.matcher(name).replaceAll("");
92      }
93  
94      /**
95       * Fix names by checking if the first character is a letter or an underscore.
96       * If not, insert an underscore at the beginning of the name in order to make it valid for
97       * use in CSS or Less.
98       *
99       * @param name  the name to fix
100      * @return      valid name without number or hyphen as first character
101      */
102     protected static String fixFirstChar(String name) {
103         boolean firstCharIsAllowed = name == null || name.isEmpty() || name.charAt(0) == '_' || Character.isLetter(name.charAt(0));
104         return firstCharIsAllowed ? name : "_" + name;
105     }
106 }
