22  /**
23   * Converts ImagePacking to a JSON(P) file, with the result that each icon's details are available
24   * via a named property which encapsulates w, h, x, y and xy as pixel values as well as storing
25   * integer values for x, y, w and h in the property n.
26   *
27   * @author Robert Murphy, mklein
28   */
29  public class JsonPackingConverter extends AbstractTextConverter {
30      // JavaScript variable names may contain unicode letters, ascii digits, the dollar sign and the underscore, e.g.: "_Ï‰$0"
31      public static final Pattern CHARS_NOT_ALLOWED_IN_VARIABLES = Pattern.compile("[^\\p{L}\\d_$]");
32      public static final Collection<String> RESERVED_WORDS = Collections.unmodifiableCollection(Arrays.asList(
33              "break", "case", "catch", "continue", "debugger", "default", "delete", "do", "else",
34              "finally", "for", "function", "if", "in", "instanceof", "new", "return", "switch",
35              "this", "throw", "try", "typeof", "var", "void", "while", "with", "class", "const",
36              "enum", "export", "extends", "import", "super", "implements", "interface", "let",
37              "package", "private", "protected", "public", "static", "yield", "null", "true",
38              "false", "NaN", "Infinity", "undefined", "eval", "arguments", "int", "byte", "char",
39              "goto", "long", "final", "float", "short", "double", "native", "throws", "boolean",
40              "abstract", "volatile", "transient", "synchronized"));
41  
42      protected static final int IMAGE_PROPERTY_COUNT = 6;
43      protected static final int RAW_NUMBER_COUNT = 4;
44  
45      final String jsonpVar;
46  
47      /**
48       * Create a JSON converter with output file json and optional JSONP variable jsonpVar.
49       * @param json     the output JSON file to write to
50       * @param jsonpVar optional JSONP variable name
51       */
52      public JsonPackingConverter(Path json, String jsonpVar) {
53          super(json, "JSON");
54          this.jsonpVar = fixJsonpVar(jsonpVar);
55      }
56  
57      /**
58       *Create output JSON string based on an ImagePacking.
59       *
60       * @param imageList     the list of images - must not be null
61       * @param imagePacking  the ImagePacking to convert - must not be null
62       * @param log           the log object to use
63       * @return              String containing the JSON file contents
64       * @throws MojoExecutionException
65       */
66      @Override
67      protected String createOutput(List<NamedImage> imageList, ImagePacking imagePacking, Log log) throws MojoExecutionException {
68          Map<String, Object> map = buildOutputMap(imageList, imagePacking);
69  
70          try {
71              ObjectMapper mapper = new ObjectMapper();
72              mapper.configure(SerializationConfig.Feature.INDENT_OUTPUT, true);
73  
74              String out = mapper.writeValueAsString(map);
75              return jsonpVar == null ? out : jsonpVar + " = " + out;
76          } catch (IOException e) {
77              throw new MojoExecutionException("Couldn't generate JSON data", e);
78          }
79      }
80  
81      /**
82       * Builds the output structure
83       * @param imageList a list of images, defines the order of the entries in the map
84       * @param imagePacking a packing of the images in the list
85       * @return a map with an entry for each image
86       */
87      protected Map<String, Object> buildOutputMap(List<NamedImage> imageList, ImagePacking imagePacking) {
88          Map<String, Object> map = new LinkedHashMap<>(imageList.size());
89          for (NamedImage n : imageList) {
90              Point position = imagePacking.getPosition(n);
91  
92              Map<String, Object> props = new LinkedHashMap<>(IMAGE_PROPERTY_COUNT);
93              int x = position.x;
94              int y = position.y;
95              int width = n.getWidth();
96              int height = n.getHeight();
97  
98              String xStr = intToPixel(-x);
99              String yStr = intToPixel(-y);
100 
101             props.put("x", xStr);
102             props.put("y", yStr);
103             props.put("w", intToPixel(width));
104             props.put("h", intToPixel(height));
105             props.put("xy", xStr + " " + yStr);
106 
107             Map<String, Integer> numbers = new LinkedHashMap<>(RAW_NUMBER_COUNT);
108             numbers.put("x", x);
109             numbers.put("y", y);
110             numbers.put("w", width);
111             numbers.put("h", height);
112 
113             props.put("n", numbers);
114 
115             map.put(n.getName(), props);
116         }
117         return map;
118     }
119 
120     /**
121      * Remove invalid characters from JavaScript variable and avoid leading numbers or reserved words
122      * by prefixing them with an underscore
123      *
124      * @param jsonpVar  the jsonpVar to fix
125      * @return          the fixed jsonpVar
126      */
127     protected static String fixJsonpVar(String jsonpVar) {
128         if (StringUtils.isNotEmpty(jsonpVar)) {
129             String newVar = CHARS_NOT_ALLOWED_IN_VARIABLES.matcher(jsonpVar).replaceAll("");
130             boolean needsPrefix = newVar.isEmpty() || hasLeadingNumber(newVar) || RESERVED_WORDS.contains(newVar);
131             return needsPrefix ? "_" + newVar : newVar;
132         }
133         return jsonpVar;
134     }
135 
136     /**
137      * Check if a string starts with an arabic [0-9] digit character
138      *
139      * @param str   the string to check
140      * @return      whether the string starts with [0-9]
141      */
142     private static boolean hasLeadingNumber(String str) {
143         return str.charAt(0) >= '0' && str.charAt(0) <= '9';
144     }
