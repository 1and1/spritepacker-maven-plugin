10  import java.awt.Point;
11  import java.io.File;
12  import java.util.HashMap;
13  import java.util.Map;

15  /**
16   * Converts ImagePacking to a JSON(P) file, with the result that each icon's details are available
17   * via a named property which encapsulates w, h, x, y and xy as pixel values as well as storing
18   * integer values for x, y, w and h in the property n.
19   *
20   * @author Robert Murphy, mklein
21   */
22  public class JsonPackingConverter extends AbstractTextConverter {
23      private final String jsonpVar;
24  
25      /**
26       * Create a JSON converter with output file json and optional JSONP variable jsonpVar.
27       *
28       * @param json     the output JSON file to write to
29       * @param jsonpVar optional JSONP variable name
30       */
31      public JsonPackingConverter(File json, String jsonpVar) {
32          super(json, "JSON");
33          this.jsonpVar = jsonpVar;
34      }
35  
36      /**
37       *Create output JSON string based on an ImagePacking.
38       *
39       * @param imagePacking  the ImagePacking to convert
40       * @param log           the log object to use
41       * @return              String containing the JSON file contents
42       * @throws MojoExecutionException
43       */
44      @Override
45      protected String createOutput(ImagePacking imagePacking, Log log) throws MojoExecutionException {
46          ObjectMapper mapper = new ObjectMapper();
47          mapper.configure(SerializationConfig.Feature.INDENT_OUTPUT, true);
48  
49          Map<String, Object> map = new HashMap<>();
50          for (NamedImage n : imagePacking.getImages()) {
51              Point position = imagePacking.getPosition(n);
52  
53              Map<String, Object> props = new HashMap<>();
54              int x = position.x;
55              int y = position.y;
56              int width = n.getWidth();
57              int height = n.getHeight();
58  
59              String xStr = intToPixel(-x);
60              String yStr = intToPixel(-y);
61  
62              props.put("x", xStr);
63              props.put("y", yStr);
64              props.put("w", intToPixel(width));
65              props.put("h", intToPixel(height));
66              props.put("xy", xStr + " " + yStr);
67  
68              Map<String, Integer> numbers = new HashMap<>();
69              numbers.put("x", x);
70              numbers.put("y", y);
71              numbers.put("w", width);
72              numbers.put("h", height);
73  
74              props.put("n", numbers);
75  
76              map.put(n.getName(), props);
77          }
78  
79          // Generate json representation of map object
80          String out;
81          try {
82              out = mapper.writeValueAsString(map);
83          } catch (Exception e) {
84              throw new MojoExecutionException("Couldn't generate JSON data", e);
85          }
86  
87          // If user has passed in a variable to wrap this in, append it to the front
88          if (jsonpVar != null) {
89              out = jsonpVar + " = " + out;
90          }
91  
92          return out;
93      }
94  }
