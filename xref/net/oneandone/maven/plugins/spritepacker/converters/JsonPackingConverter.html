22  /**
23   * Converts ImagePacking to a JSON(P) file, with the result that each icon's details are available
24   * via a named property which encapsulates w, h, x, y and xy as pixel values as well as storing
25   * integer values for x, y, w and h in the property n.
26   *
27   * @author Robert Murphy, mklein
28   */
29  public class JsonPackingConverter extends AbstractTextConverter {
30      public static final Pattern CHARS_NOT_ALLOWED_IN_VARIABLES = Pattern.compile("[^\\p{L}\\d_$]");
31      public static final Collection<String> RESERVED_WORDS = Collections.unmodifiableCollection(Arrays.asList(
32              "break", "case", "catch", "continue", "debugger", "default", "delete", "do", "else",
33              "finally", "for", "function", "if", "in", "instanceof", "new", "return", "switch",
34              "this", "throw", "try", "typeof", "var", "void", "while", "with", "class", "const",
35              "enum", "export", "extends", "import", "super", "implements", "interface", "let",
36              "package", "private", "protected", "public", "static", "yield", "null", "true",
37              "false", "NaN", "Infinity", "undefined", "eval", "arguments", "int", "byte", "char",
38              "goto", "long", "final", "float", "short", "double", "native", "throws", "boolean",
39              "abstract", "volatile", "transient", "synchronized"));
40  
41      protected static final int IMAGE_PROPERTY_COUNT = 6;
42      protected static final int RAW_NUMBER_COUNT = 4;
43  
44      final String jsonpVar;
45  
46      /**
47       * Create a JSON converter with output file json and optional JSONP variable jsonpVar.
48       * @param json     the output JSON file to write to
49       * @param jsonpVar optional JSONP variable name
50       */
51      public JsonPackingConverter(Path json, String jsonpVar) {
52          super(json, "JSON");
53          this.jsonpVar = fixJsonpVar(jsonpVar);
54      }
55  
56      /**
57       *Create output JSON string based on an ImagePacking.
58       *
59       * @param imageList     the list of images - must not be null
60       * @param imagePacking  the ImagePacking to convert - must not be null
61       * @param log           the log object to use
62       * @return              String containing the JSON file contents
63       * @throws MojoExecutionException
64       */
65      @Override
66      protected String createOutput(List<NamedImage> imageList, ImagePacking imagePacking, Log log) throws MojoExecutionException {
67          Map<String, Object> map = buildOutputMap(imageList, imagePacking);
68  
69          try {
70              ObjectMapper mapper = new ObjectMapper();
71              mapper.configure(SerializationConfig.Feature.INDENT_OUTPUT, true);
72  
73              String out = mapper.writeValueAsString(map);
74              return jsonpVar == null ? out : jsonpVar + " = " + out;
75          } catch (IOException e) {
76              throw new MojoExecutionException("Couldn't generate JSON data", e);
77          }
78      }
79  
80      /**
81       * Builds the output structure
82       * @param imageList a list of images, defines the order of the entries in the map
83       * @param imagePacking a packing of the images in the list
84       * @return a map with an entry for each image
85       */
86      protected Map<String, Object> buildOutputMap(List<NamedImage> imageList, ImagePacking imagePacking) {
87          Map<String, Object> map = new LinkedHashMap<>(imageList.size());
88          for (NamedImage n : imageList) {
89              Point position = imagePacking.getPosition(n);
90  
91              Map<String, Object> props = new LinkedHashMap<>(IMAGE_PROPERTY_COUNT);
92              int x = position.x;
93              int y = position.y;
94              int width = n.getWidth();
95              int height = n.getHeight();
96  
97              String xStr = intToPixel(-x);
98              String yStr = intToPixel(-y);
99  
100             props.put("x", xStr);
101             props.put("y", yStr);
102             props.put("w", intToPixel(width));
103             props.put("h", intToPixel(height));
104             props.put("xy", xStr + " " + yStr);
105 
106             Map<String, Integer> numbers = new LinkedHashMap<>(RAW_NUMBER_COUNT);
107             numbers.put("x", x);
108             numbers.put("y", y);
109             numbers.put("w", width);
110             numbers.put("h", height);
111 
112             props.put("n", numbers);
113 
114             map.put(n.getName(), props);
115         }
116         return map;
117     }
118 
119     /**
120      * Remove invalid characters from JavaScript variable and avoid leading numbers or reserved words
121      * by prefixing them with an underscore
122      *
123      * @param jsonpVar  the jsonpVar to fix
124      * @return          the fixed jsonpVar
125      */
126     protected static String fixJsonpVar(String jsonpVar) {
127         if (StringUtils.isNotEmpty(jsonpVar)) {
128             String newVar = CHARS_NOT_ALLOWED_IN_VARIABLES.matcher(jsonpVar).replaceAll("");
129             boolean needsPrefix = newVar.isEmpty() || hasLeadingNumber(newVar) || RESERVED_WORDS.contains(newVar);
130             return needsPrefix ? "_" + newVar : newVar;
131         }
132         return jsonpVar;
133     }
134 
135     /**
136      * Check if a string starts with an arabic [0-9] digit character
137      *
138      * @param str   the string to check
139      * @return      whether the string starts with [0-9]
140      */
141     private static boolean hasLeadingNumber(String str) {
142         return str.charAt(0) >= '0' && str.charAt(0) <= '9';
143     }
