8   import java.awt.Point;
9   import java.io.File;

11  /**
12   * Converts ImagePacking to a CSS file, with the result that each icon gets its own class containing
13   * background-position, width and height of the icon.
14   *
15   * @author mklein
16   */
17  public class CssPackingConverter extends AbstractTextConverter {
18      private final String cssPrefix;
19  
20      /**
21       * Create a CSS converter with output file css and class prefix cssPrefix.
22       *
23       * @param css       the output CSS file to write to
24       * @param cssPrefix the CSS class prefix for each icon class
25       */
26      public CssPackingConverter(File css, String cssPrefix) {
27          super(css, "CSS");
28          this.cssPrefix = cssPrefix;
29      }
30  
31      /**
32       * Create output CSS string based on an ImagePacking.
33       *
34       * @param imagePacking  the ImagePacking to convert
35       * @param log           the log object to use
36       * @return              String containing the CSS file contents
37       * @throws MojoExecutionException
38       */
39      @Override
40      protected String createOutput(ImagePacking imagePacking, Log log) throws MojoExecutionException {
41          String prefix = this.cssPrefix.equals("") ? "" : this.cssPrefix + "-";
42          StringBuilder sb = new StringBuilder("/* this file is generated by the sprite packer. don't make any changes in here! */\n");
43          for (NamedImage image : imagePacking.getImages()) {
44              String name = image.getName();
45              Point position = imagePacking.getPosition(image);
46              String x = intToPixel(-position.x);
47              String y = intToPixel(-position.y);
48              String width = intToPixel(image.getWidth());
49              String height = intToPixel(image.getHeight());
50              sb.append(".").append(prefix).append(name).append("{background-position:").append(x).append(" ").append(y).append(";")
51                      .append("width:").append(width).append(";height:").append(height).append(";}\n");
52          }
53          return sb.toString();
54      }
55  }
