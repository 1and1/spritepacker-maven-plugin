1   package net.oneandone.maven.plugins.spritepacker;
2   
3   import java.awt.Dimension;
4   import java.awt.Point;
5   import java.util.Collections;
6   import java.util.Map;
7   
8   /**
9    * Holds the result of the PackGrowing. Contains a Map of images to positions
10   * as well as the outer dimensions of the spritesheet.
11   *
12   * @author ssiegler
13   */
14  public class ImagePacking {
15      private final Map<NamedImage, Point> positionMap;
16      private final int width;
17      private final int height;
18  
19      /**
20       * Create an ImagePacking with the given dimension and position map data.
21       *
22       * @param dimension     the outer dimensions of the spritesheet
23       * @param positionMap   a map of images to positions
24       */
25      public ImagePacking(Dimension dimension, Map<NamedImage, Point> positionMap) {
26          width = dimension.width;
27          height = dimension.height;
28          this.positionMap = Collections.unmodifiableMap(positionMap);
29      }
30  
31      /**
32       * Get the position of the specified image within the spritesheet.
33       *
34       * @param image the image to get the position of
35       * @return      the position of the given image
36       */
37      public Point getPosition(NamedImage image) {
38          return positionMap.get(image);
39      }
40  
41      public int getWidth() {
42          return width;
43      }
44  
45      public int getHeight() {
46          return height;
47      }
48  
49      @Override
50      public String toString() {
51          return "ImagePacking{" +
52                 "positionMap=" + positionMap +
53                 ", width=" + width +
54                 ", height=" + height +
55                 '}';
56      }
57  }
