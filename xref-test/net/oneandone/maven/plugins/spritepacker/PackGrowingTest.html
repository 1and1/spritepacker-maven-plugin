32      @Test
33      public void testMappingWithoutImages() throws Exception {
34          ImagePacking fit = PackGrowing.fit(Collections.<NamedImage>emptyList(), 3);
35          errorCollector.checkThat(fit.getImages(), is(empty()));
36          errorCollector.checkThat(fit.getHeight(), is(0));
37          errorCollector.checkThat(fit.getWidth(), is(0));
38      }

40      @Test
41      public void testMappingForSingleImage() throws Exception {
42          int padding = 5;
43          NamedImage image = new NamedImage(new BufferedImage(30, 20, BufferedImage.TYPE_INT_ARGB), "Bild");
44          ImagePacking fit = PackGrowing.fit(Collections.singletonList(image), padding);
45          errorCollector.checkThat(fit.getImages(), contains(image));
46          errorCollector.checkThat(fit.getHeight(), is(image.getHeight() + 2 * padding));
47          errorCollector.checkThat(fit.getWidth(), is(image.getWidth() + 2 * padding));
48          errorCollector.checkThat(fit.getPosition(image), is(new Point(padding, padding)));
49      }

51      @Test
52      public void testMappingForTwoImagesInARow() throws Exception {
53          int padding = 8;
54          int width = 43;
55          int height = 34;
56          List<NamedImage> images = Arrays.asList(
57                  new NamedImage(new BufferedImage(width, height, BufferedImage.TYPE_INT_ARGB), "Bild1"),
58                  new NamedImage(new BufferedImage(width, height, BufferedImage.TYPE_INT_ARGB), "Bild2"));
59          ImagePacking fit = PackGrowing.fit(images, padding);
60          errorCollector.checkThat(new HashSet<>(fit.getImages()), is(new HashSet<>(images)));
61          errorCollector.checkThat(fit.getHeight(), is(height + 2 * padding));
62          errorCollector.checkThat(fit.getWidth(), is(2 * width + 3 * padding));
63          errorCollector.checkThat(fit.getPosition(images.get(0)), is(new Point(padding, padding)));
64          errorCollector.checkThat(fit.getPosition(images.get(1)), is(new Point(2 * padding + width, padding)));
65      }

67      @Test
68      public void testMappingForTwoImagesInAColumn() throws Exception {
69          int padding = 8;
70          int width = 43;
71          List<NamedImage> images = Arrays.asList(
72                  new NamedImage(new BufferedImage(width, 34, BufferedImage.TYPE_INT_ARGB), "Bild1"),
73                  new NamedImage(new BufferedImage(width, 35, BufferedImage.TYPE_INT_ARGB), "Bild2"));
74          ImagePacking fit = PackGrowing.fit(images, padding);
75          errorCollector.checkThat(new HashSet<>(fit.getImages()), is(new HashSet<>(images)));
76          errorCollector.checkThat(fit.getHeight(), is(images.get(0).getHeight() + images.get(1).getHeight() + 3 * padding));
77          errorCollector.checkThat(fit.getWidth(), is(width + 2 * padding));
78          errorCollector.checkThat(fit.getPosition(images.get(0)), is(new Point(padding, padding)));
79          errorCollector.checkThat(fit.getPosition(images.get(1)), is(new Point(padding, 2 * padding + images.get(0).getHeight())));
80      }

82      @Test
83      public void testMappingForManyImages() throws Exception {
84          int size = 500;
85          List<NamedImage> images = new ArrayList<>(size);
86          for (int i = 0; i < size; i++) {
87              ThreadLocalRandom random = ThreadLocalRandom.current();
88              images.add(new NamedImage(new BufferedImage(random.nextInt(1, 1024), random.nextInt(1, 1024), BufferedImage.TYPE_INT_ARGB), "Bild" + i));
89          }
90  
91          ImagePacking fit = PackGrowing.fit(images, 2);
92          errorCollector.checkThat(fit.getImages(), containsInAnyOrder(images.toArray()));
93          Rectangle boundingBox = new Rectangle(fit.getWidth(), fit.getHeight());
94  
95          List<Rectangle> rects = new ArrayList<>(size);
96  
97          for (NamedImage image : images) {
98              Point position = fit.getPosition(image);
99              rects.add(new Rectangle(position.x, position.y, image.getWidth(), image.getHeight()));
100             errorCollector.checkThat(position, is(notNullValue()));
101             errorCollector.checkThat(boundingBox.contains(position), is(true));
102         }
103 
104         int overlappingCount = 0;
105         for (int i = 0; i < rects.size(); i++) {
106             for (int j = 0; j < rects.size(); j++) {
107                 if (i != j) {
108                     if (rects.get(i).contains(rects.get(j).getLocation())) {
109                         overlappingCount++;
110                     }
111                 }
112             }
113         }
114 
115         errorCollector.checkThat("Overlapping images", overlappingCount, is(0));
116     }
