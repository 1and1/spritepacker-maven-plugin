108     @Test
109     public void testMappingForManyImages() throws Exception {
110         int size = 500;
111         List<NamedImage> images = new ArrayList<>(size);
112         for (int i = 0; i < size; i++) {
113             ThreadLocalRandom random = ThreadLocalRandom.current();
114             images.add(new NamedImage(new BufferedImage(random.nextInt(1, 1024), random.nextInt(1, 1024), BufferedImage.TYPE_INT_ARGB), "Bild" + i));
115         }
116 
117         ImagePacking fit = PackGrowing.fit(images, 2);
118         Rectangle boundingBox = new Rectangle(fit.getWidth(), fit.getHeight());
119 
120         List<Rectangle> rects = new ArrayList<>(size);
121 
122         for (NamedImage image : images) {
123             Point position = fit.getPosition(image);
124             errorCollector.checkThat(position, is(notNullValue()));
125 
126             rects.add(new Rectangle(position.x, position.y, image.getWidth(), image.getHeight()));
127             errorCollector.checkThat(boundingBox.contains(position), is(true));
128         }
129 
130         int overlappingCount = 0;
131         for (int i = 0; i < rects.size(); i++) {
132             for (int j = 0; j < rects.size(); j++) {
133                 if (i != j) {
134                     if (rects.get(i).contains(rects.get(j).getLocation())) {
135                         overlappingCount++;
136                     }
137                 }
138             }
139         }
140 
141         errorCollector.checkThat("Overlapping images", overlappingCount, is(0));
142     }
